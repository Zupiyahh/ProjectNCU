void state(int &flag) {
  if (digitalRead(Lswitch) == HIGH) {
    Serial.println("State:Free");
    NodeSerial.println("State:Free");
  }
  else if ( (digitalRead(Lswitch) == LOW) && (flag == 0) ) {
    Serial.println("State:Busy");
    NodeSerial.println("State:Busy");
    flag = 1;

    //ตั้งเวลาเกิน
    for (int i = 15; i >= 0; i -- ) {
      Serial.print("\tcountdown:");
      Serial.println(i);
      if ( (digitalRead(Lswitch) == HIGH) && (flag == 1) ) {
        Serial.println("State:Free");
        NodeSerial.println("State:Free");
        flag = 0;
        break ;
      }
      delay(50);
    }
    if (flag == 1) {
      Serial.println("State:Error");
      NodeSerial.println("State:Error");

    }
  }
}

void setup() {

}

void loop() {
  char Mode;
  int input;
  //รับข้อมูล mode จาก esp
  while (NodeSerial.available () > 0) {
    input = NodeSerial.parseInt();
    Mode = char(input);
    if ( Mode == '1' || Mode == '2' || Mode == '3' || Mode == '4' || Mode == '5' || Mode == '6' || Mode == '7' || Mode == '8' ) {
      Serial.print(" Mode is ");
      Serial.print(Mode);
      Serial.print("\n");
      actionMode(Mode);
    }
  }
}
